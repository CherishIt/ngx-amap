直接设置属性

<amap-polyline [path]="lineArr" [strokeColor]="'#3366FF'" [strokeOpacity]="1" [strokeWeight]="5" [strokeStyle]="'solid'" [strokeDasharray]="[10, 5]"> </amap-polyline> .demo-map {{'{'}} height: 400px; } lineArr = [ [116.368904, 39.913423], [116.382122, 39.901176], [116.387271, 39.912501], [116.398258, 39.904600] ];

利用options设置属性

<amap-polyline [options]="lineOptions"></amap-polyline> .demo-map {{'{'}} height: 400px; } lineOptions = {{'{'}} path: this.lineArr, // 设置线覆盖物路径 strokeColor: '#3366FF', // 线颜色 strokeOpacity: 1, // 线透明度 strokeWeight: 5, // 线宽 strokeStyle: 'solid', // 线样式 strokeDasharray: [10, 5] // 补充线样式 };
